    <h1>Missive iFrame Integration</h1>
    <p>This page demonstrates a basic integration with Missive.</p>
    <div class="log" id="log">
        <strong>Event log:</strong>
    </div>

    <script>
        // Function to log events to the page
        function logEvent(message) {
            const logContainer = document.getElementById('log');
            const newLogEntry = document.createElement('p');
            newLogEntry.textContent = message;
            logContainer.appendChild(newLogEntry);
        }

        // Listening to conversation changes in Missive
        Missive.on('change:conversations', (ids) => {
            logEvent('Conversations changed: ' + JSON.stringify(ids));
        });

        // Example of fetching and logging conversations
        Missive.on('ready', () => {
            logEvent('Missive API is ready.');
            Missive.fetchConversations().then(conversations => {
                logEvent('Fetched conversations: ' + JSON.stringify(conversations));
            });
        });
    </script>
